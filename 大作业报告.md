/**
 * 用户登录
 * @param uname 用户名
 * @param upwd 密码
 * @param session 会话
 * @return 登录结果页面
 */
@PostMapping("/login.do")
public String login(String uname, String upwd, HttpSession session) {
    User user = userService.findByNameAndPwd(uname, upwd);
    if (user != null && user.getUstate() == 1) {
        session.setAttribute("curruser", user);
        return "redirect:/";
    }
    return "login";
}

/**
 * 用户注销
 * @param uid 用户ID
 * @param session 会话
 * @return 注销结果
 */
@GetMapping("/deleteAccount/{uid}")
public String deleteAccount(@PathVariable int uid, HttpSession session) {
    User currUser = (User) session.getAttribute("curruser");
    if (currUser == null) {
        return "redirect:/error";
    }

    // 管理员可以注销普通用户，普通用户只能注销自己
    if (currUser.getLevel() == 0 || currUser.getUid() == uid) {
        if (currUser.getLevel() == 0 && currUser.getUid() == uid) {
            return "redirect:/error"; // 管理员不能注销自己
        }
        userService.deleteUser(uid);
        if (currUser.getUid() == uid) {
            session.invalidate();
            return "redirect:/login";
        }
        return "redirect:/host";
    }
    return "redirect:/error";
}
```

/**
 * 分页查询帖子
 * @param page 页码
 * @return 分页结果
 */
public Page<Post> findByPage(int page) {
    Page<Post> pageResult = new Page<>();
    pageResult.setPage(page);

    // 查询总记录数
    int total = postMapper.count();
    pageResult.setTotal(total);

    // 计算偏移量
    int offset = (page - 1) * pageResult.getSize();
    List<Post> posts = postMapper.findByPage(offset, pageResult.getSize());
    pageResult.setRecords(posts);

    return pageResult;
}
```

/**
 * 保存回复
 * @param reply 回复对象
 * @return void
 */
@Override
public void save(Reply reply) {
    replyMapper.save(reply);

    // 更新帖子的最后回复时间
    postMapper.updateLastReplyTime(reply.getPid());
}

/**
 * 置顶回复
 * @param rid 回复ID
 * @param pid 帖子ID
 * @param uid 用户ID
 * @return 是否成功
 */
public boolean stickyReply(int rid, int pid, int uid) {
    Post post = postMapper.findById(pid);

    // 只有帖子创建者可以置顶回复
    if (post.getUid() != uid) {
        return false;
    }

    // 取消该帖子其他回复的置顶状态
    replyMapper.unstickyAll(pid);

    // 置顶指定回复
    replyMapper.sticky(rid);

    // 发放积分奖励
    if ("问题".equals(post.getCategory()) && post.getPrize() > 0) {
        Reply reply = replyMapper.findById(rid);
        userMapper.addScore(reply.getUid(), post.getPrize());
    }

    return true;
}
```

/**
 * 全局异常处理器
 */
@ControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(Exception.class)
    public String handleException(Exception e, Model model) {
        model.addAttribute("error", e.getMessage());
        return "error";
    }
}

/**
 * OSS工具类
 */
@Component
public class OSSUtil {

    @Value("${oss.endpoint}")
    private String endpoint;

    @Value("${oss.bucket-name}")
    private String bucketName;

    @Value("${oss.access-key-id}")
    private String accessKeyId;

    @Value("${oss.access-key-secret}")
    private String accessKeySecret;

    /**
     * 上传文件到OSS
     * @param file 文件
     * @param fileName 文件名
     * @return 文件URL
     */
    public String uploadFile(MultipartFile file, String fileName) throws IOException {
        // 创建OSS客户端
        OSS ossClient = OSSClientBuilder.create()
            .endpoint(endpoint)
            .credentialsProvider(new DefaultCredentialProvider(accessKeyId, accessKeySecret))
            .build();

        try {
            // 上传文件
            ossClient.putObject(bucketName, fileName, file.getInputStream());

            // 返回文件URL
            return "https://" + bucketName + "." + endpoint + "/" + fileName;
        } finally {
            ossClient.shutdown();
        }
    }
}
```

WantedBy=multi-user.target

sudo systemctl start bbs
